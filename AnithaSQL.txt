CREATE TABLE BANKTRANSACTION(ACNO INT FOREIGN KEY REFERENCES BANKMASTER(ACNO),TTYPE CHAR(1) CHECK(TTYPE='W' OR TTYPE='D'),TDATE DATETIME NOT NULL,AMT MONEY NOT NULL)
